#include <stdio.h>
#include <iostream>
#include <string>
#include <math.h>
#include <cmath>
using namespace std;

int main()
{
	setlocale(LC_ALL, "Russian");
	int sec; // время горения зеленого сигнала светофора (25)
	int h; // количевство полос (4)
	int f; // количевство перекрестков (2)
	int j; //пересекаемая дорога
	int l; //полосность пересекаемой улицы

	cout << "Время горения зеленого сигнала светофора в секундах:";
	cin >> sec;
	cout << "Количевство полос:";
	cin >> h;
	cout << "Количевство перекрестков:";
	cin >> f;
	cout << "Количевство полос на пересекаемой улице(-ах): ";
	cin >> l;

	int n = sec - 10; //количевство авто которое успеет проехать на зел.сигн по одной полосе
	if (sec < 10)
	{
		cout << "Ни какой автомобиль не проедет" << endl;
		return 0;
	}

	int d = n * h; //количевство авто которое успеет проехать по всем полосам на одном перекрестке
	int o = (n - 10) * h; // количевство авто которое успеет проехать на зел. сигн по одн. полосе через 2 перекр.

	if (f = 1)
	{
		d = d + (f * (n * l));
		cout << "Количевство авто которое успеет проехать на зеленый сигнал за минуту: " << endl << d;
	}
	if (f > 1)
	{
		o = o + (f * (n * l));
		cout << "Количевство авто которое успеет проехать на зеленый сигнал за минуту: " << endl << o;
	}
	return 0;
}
